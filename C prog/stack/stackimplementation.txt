#include<stdio.h>
#include<stdlib.h>
#define MAX 5
int top=-1;
int stack_arr[MAX];
void push(int data);
int pop();
int print();
int peek();
int main()
{
    int choice , data;
    while(1)
    {
        printf("\n1. Push\n");
        printf("2. Pop\n");
        printf("3. Print Stack\n");
        printf("4. Print Top Stack.\n");
        printf("5. Quit\n");
        printf("Enter Choice:");
        scanf("%d",&choice);
        switch(choice)
        {
            case 1:
                printf("Enter the element to be pushed:\n");
                scanf("%d",&data);
                push(data);
                break;
            case 2:
                data=pop();
                printf("Deleted Element:%d",data);
                break;
            case 3:
                print();
                break;
            case 4:
                printf("Topmost element = %d\n",peek());
                break;
            case 5:
                exit(1);
            default:
                printf("Wrong Choice\n");
        }
    }
    return 0;
}

void push(int data)
{
    if(top==MAX-1)
    {
        printf("Stack Overflow\n");
    }
    top=top+1;
    stack_arr[top]=data;
}
 int pop()
 {
     int value;
     if(top==-1)
     {
         printf("Stack Underflow\n");
     }
     value=stack_arr[top];
     top=top-1;
     return value;
 }
 
 int print()
 {
     for(int i=top;i>=0;i--)
     {
         printf("%d\t",stack_arr[i]);
     }
 }
 int peek()
 {
     if(top==-1)
     {
         printf("Stack Underflow\n");
         return -1;
     }
     return stack_arr[top];
 }
