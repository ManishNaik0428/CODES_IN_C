#include<stdio.h>
#define MAX 5
int stack_arr[MAX];
int top=-1;
void push(int);
int pop();
void print();
int main()
{
    push(10);
    push(20);
    push(30);
    push(40);
    push(50);
    print();
    int data=pop();
    print();
    return 0;
}
void push(int data)
{
 if(top==MAX-1)
 {
     printf("Stack Overflow\n");
 }
    top=top+1;
    stack_arr[top]=data;
}
int pop()
{
    int value;
    if(top==-1)
    {
        printf("Stack Underflow");
    }
   value=stack_arr[top];
   top=top-1;
   return value;
}
void print()
{
    printf("\nStack elements are:\n");
    for (int i = top; i>=0; i--) {
        printf("%d\t", stack_arr[i]);
    }
}
