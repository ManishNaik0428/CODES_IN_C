#include<stdio.h>
 int main()
 {
     int arr[15],n,i,j,k;
     printf("Enter number:");
     scanf("%d",&n);
     for(i=0;i<n;i++)
     {
         printf("number %d=",i+1);
         scanf("%d",&arr[i]);
     }
     for(i=0;i<n;i++)
     {
        for(j=i+1;j<n;j++)
        {
            if(arr[i]==arr[j])
            {
                for(k=j;k<n;k++)
                {
                    arr[k]=arr[k+1];
                }
                k--;
                n--;
            }
        }
     }
     for(i=0;i<n;i++)
     {
         printf("[%d]",arr[i]);
     }
     return 0;
 }
