#include<stdio.h>
int main()
{
    int arr[100],arr1[100],arr2[100],n,n1,n2,i,j,k;
    printf("enter number to enter in first arrray:");
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        printf("Arr[%d]:",i);
        scanf("%d",&arr[i]);
    }
    printf("\nEnter number to enter in second array:");
    scanf("%d",&n1);
    for(j=0;j<n1;j++)
    {
        printf("Arr1[%d]:",j);
        scanf("%d",&arr1[j]);
    }
    n2=n+n1;
    for(i=0;i<n;i++)
    {
        arr2[i]=arr[i];
    }
    for(j=0;j<n1;j++)
    {
        arr2[i]=arr1[j];
        i++;
    }
    for(i=0;i<n2;i++)
    {
        for(k=0;k<n2-1;k++)
        {
            if(arr2[k]>arr2[k+1])
            {
                j=arr2[k+1];
                arr2[k+1]=arr2[k];
                arr2[k]=j;
            }
        }
    }
printf("Ascending Order :\n");
for(i=0;i<n2;i++)
{
    printf("%d\t",arr2[i]);
}
printf("\n******************************************************\n");
for(i=0;i<n2;i++)
{
    for(k=0;k<n2-1;k++)
    {
        if(arr2[k]<arr2[k+1])
        {
            j=arr2[k+1];
            arr2[k+1]=arr2[k];
            arr2[k]=j;
        }
    }
}
printf("Descending Order:\n");
for(i=0;i<n2;i++)
{
    printf("%d\t",arr2[i]);
}
return 0;
}
