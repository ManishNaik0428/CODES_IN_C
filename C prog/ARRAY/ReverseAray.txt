#include<stdio.h>
 int main()
 {
     int arr[15],n,i;
     printf("Enter number:");
     scanf("%d",&n);
     for(i=0;i<n;i++)
     {
         printf("number %d=",i+1);
         scanf("%d",&arr[i]);
     }
    printf("Original Array \n");
    for(i=0;i<n;i++)
    {
        printf("%d\t",arr[i]);
    }
    printf("\nReverse of an array \n");
    //Reverse array
    for(i=n-1;i>=0;i--)
    {
        printf("%d\t",arr[i]);
    }
     return 0;
 }
