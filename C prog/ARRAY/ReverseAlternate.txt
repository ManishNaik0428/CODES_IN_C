#include<stdio.h>
int main()
{
  int arr[100],i,n,temp;
  int j=n;
  printf("Enter the number:");
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
      printf("Arr[%d]:",i);
      scanf("%d",&arr[i]);

  {
      printf("%d\t",arr[i]);
  }
    return 0;
}
